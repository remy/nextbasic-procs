#autoline 10

DEFPROC loadAssets()
  LAYER 2,1

  ; shadow L2 ends at 14, so we start at 15
  LOAD "assets/game.pal" BANK 15
  SPRITE PALETTE 0 BANK 15

  LOAD "assets/sprites.spr" BANK 16
  SPRITE BANK 16
  SPRITE PRINT 1
  ; border over sprites 1 (true)
  SPRITE BORDER 1

  LOAD "assets/tiles.map" BANK 17
  TILE BANK 17
  ; bank, offset, tile width, tile size (16 | 8)
  TILE DIM 17,0,10,16

ENDPROC
