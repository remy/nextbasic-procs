DEFPROC onAnyKey()
  REPEAT : REPEAT UNTIL INKEY$ =""
  REPEAT : REPEAT UNTIL INKEY$ <> "" OR ( IN 31=16)
  REPEAT : REPEAT UNTIL INKEY$ =""
ENDPROC

DEFPROC onAnyKeyWithSprite()
  REPEAT : SPRITE MOVE INT : REPEAT UNTIL INKEY$ =""
  REPEAT : SPRITE MOVE INT : REPEAT UNTIL INKEY$ <> "" OR ( IN 31=16)
  REPEAT : SPRITE MOVE INT : REPEAT UNTIL INKEY$ =""
ENDPROC
